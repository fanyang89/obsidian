python-prctl >= 1.5.0
python-psutil >= 5.4.3
python-snappy
python2-argcomplete

rpmlib(FileDigests) <= 4.6.0-1

rpmlib(PayloadFilesHavePrefix) <= 4.0-1
rpmlib(PayloadIsXz) <= 5.2-1
```
